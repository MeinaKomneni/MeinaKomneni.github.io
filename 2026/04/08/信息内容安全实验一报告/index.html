/*
 * 实验1(第一部分) - 基于 libpcap 的 HTTP 被动捕包 (C 语言版本)
 * 使用 pcap_loop + 回调函数捕获 HTTP 数据包并分析协议
 *
 * 编译: gcc -o http_capture http_capture.c -lpcap
 * 运行: sudo ./http_capture [接口名]
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>
#include <pcap.h>
#include <netinet/ip.h>
#include <netinet/ip6.h>
#include <netinet/tcp.h>
#include <netinet/if_ether.h>
#include <arpa/inet.h>

#define SNAP_LEN 65535
#define MAX_PAYLOAD_PRINT 2000
#define SEPARATOR "======================================================================"

/* 获取当前时间字符串 HH:MM:SS */
static void get_timestr(char *buf, size_t len)
{
    time_t now = time(NULL);
    struct tm *t = localtime(&now);
    strftime(buf, len, "%H:%M:%S", t);
}

/* 检查 payload 是否以指定前缀开头 */
static int starts_with(const char *s, const char *prefix)
{
    return strncmp(s, prefix, strlen(prefix)) == 0;
}

/* 打印 HTTP 请求头部各行 */
static void print_header_lines(const char *headers, int header_len)
{
    const char *p = headers;
    const char *end = headers + header_len;

    /* 跳过请求行/状态行 */
    while (p < end && *p != '\r' && *p != '\n')
        p++;
    if (p + 2 <= end && p[0] == '\r' && p[1] == '\n')
        p += 2;

    while (p < end) {
        if (*p == '\r' || *p == '\n')
            break;
        const char *line_end = p;
        while (line_end < end && *line_end != '\r' && *line_end != '\n')
            line_end++;
        printf("    %.*s\n", (int)(line_end - p), p);
        p = line_end;
        if (p + 2 <= end && p[0] == '\r' && p[1] == '\n')
            p += 2;
        else if (p < end)
            p++;
    }
}

/* 从 headers 中提取 Host 值 */
static void extract_host(const char *headers, int header_len, char *host, size_t host_size)
{
    host[0] = '\0';
    const char *p = headers;
    const char *end = headers + header_len;

    while (p < end) {
        /* 不区分大小写匹配 "host:" */
        if ((p + 5 <= end) &&
            (p[0] == 'H' || p[0] == 'h') &&
            (p[1] == 'O' || p[1] == 'o') &&
            (p[2] == 'S' || p[2] == 's') &&
            (p[3] == 'T' || p[3] == 't') &&
            p[4] == ':') {
            const char *val = p + 5;
            while (val < end && *val == ' ')
                val++;
            const char *val_end = val;
            while (val_end < end && *val_end != '\r' && *val_end != '\n')
                val_end++;
            int len = (int)(val_end - val);
            memcpy(host, val, len);
            host[len] = '\0';
            return;
        }
        /* 跳到下一行 */
        while (p < end && *p != '\n')
            p++;
        if (p < end)
            p++;
    }
}

/* pcap_loop 回调函数：处理每个捕获的数据包 */
static void packet_callback(u_char *user, const struct pcap_pkthdr *header,
                             const u_char *packet)
{
    (void)user;
    (void)header;
    char ts[16];
    get_timestr(ts, sizeof(ts));

    const struct ether_header *eth = (const struct ether_header *)packet;
    uint16_t ether_type = ntohs(eth->ether_type);

    char src_ip[INET6_ADDRSTRLEN] = {0};
    char dst_ip[INET6_ADDRSTRLEN] = {0};
    const u_char *tcp_start = NULL;
    int ip_payload_len = 0;

    if (ether_type == ETHERTYPE_IP) {
        /* IPv4 */
        const struct ip *iph = (const struct ip *)(packet + sizeof(struct ether_header));
        if (iph->ip_p != IPPROTO_TCP)
            return;
        inet_ntop(AF_INET, &iph->ip_src, src_ip, sizeof(src_ip));
        inet_ntop(AF_INET, &iph->ip_dst, dst_ip, sizeof(dst_ip));
        int ip_hdr_len = iph->ip_hl * 4;
        tcp_start = (const u_char *)iph + ip_hdr_len;
        ip_payload_len = ntohs(iph->ip_len) - ip_hdr_len;
    } else if (ether_type == ETHERTYPE_IPV6) {
        /* IPv6 */
        const struct ip6_hdr *ip6 = (const struct ip6_hdr *)(packet + sizeof(struct ether_header));
        if (ip6->ip6_nxt != IPPROTO_TCP)
            return; /* 简化处理，不解析扩展头 */
        inet_ntop(AF_INET6, &ip6->ip6_src, src_ip, sizeof(src_ip));
        inet_ntop(AF_INET6, &ip6->ip6_dst, dst_ip, sizeof(dst_ip));
        tcp_start = (const u_char *)ip6 + 40;
        ip_payload_len = ntohs(ip6->ip6_plen);
    } else {
        return;
    }

    /* TCP 头 */
    const struct tcphdr *tcph = (const struct tcphdr *)tcp_start;
    uint16_t sport = ntohs(tcph->th_sport);
    uint16_t dport = ntohs(tcph->th_dport);
    uint8_t flags = tcph->th_flags;
    uint32_t seq = ntohl(tcph->th_seq);
    uint32_t ack = ntohl(tcph->th_ack);
    int tcp_hdr_len = tcph->th_off * 4;

    const u_char *payload = tcp_start + tcp_hdr_len;
    int payload_len = ip_payload_len - tcp_hdr_len;

    /* ===== TCP 三次握手 ===== */
    if ((flags & (TH_SYN | TH_ACK)) == TH_SYN) {
        printf("[%s] [TCP SYN]     %s:%u -> %s:%u  seq=%u\n",
               ts, src_ip, sport, dst_ip, dport, seq);
    } else if ((flags & (TH_SYN | TH_ACK)) == (TH_SYN | TH_ACK)) {
        printf("[%s] [TCP SYN-ACK] %s:%u -> %s:%u  seq=%u ack=%u\n",
               ts, src_ip, sport, dst_ip, dport, seq, ack);
    } else if ((flags & TH_ACK) && payload_len == 0 && !(flags & TH_FIN)) {
        printf("[%s] [TCP ACK]     %s:%u -> %s:%u  ack=%u\n",
               ts, src_ip, sport, dst_ip, dport, ack);
    }

    /* ===== TCP 四次挥手 ===== */
    if (flags & TH_FIN) {
        printf("[%s] [TCP FIN]     %s:%u -> %s:%u\n",
               ts, src_ip, sport, dst_ip, dport);
    }

    if (payload_len <= 0)
        return;

    /* 将 payload 当作文本处理（仅适用于明文 HTTP） */
    /* 创建一个以 null 结尾的副本 */
    int text_len = payload_len < 8192 ? payload_len : 8192;
    char text[8193];
    memcpy(text, payload, text_len);
    text[text_len] = '\0';

    /* ===== HTTP 请求报文 ===== */
    if (starts_with(text, "GET ") || starts_with(text, "POST ") ||
        starts_with(text, "HEAD ") || starts_with(text, "PUT ") ||
        starts_with(text, "DELETE ")) {

        /* 解析请求行 */
        char method[16] = {0}, path[2048] = {0}, version[16] = {0};
        const char *line_end = strstr(text, "\r\n");
        if (!line_end)
            line_end = text + strlen(text);
        int line_len = (int)(line_end - text);

        /* method */
        const char *p = text;
        int i = 0;
        while (p < line_end && *p != ' ' && i < 15)
            method[i++] = *p++;
        method[i] = '\0';
        if (p < line_end) p++; /* 跳过空格 */

        /* path */
        i = 0;
        while (p < line_end && *p != ' ' && i < 2047)
            path[i++] = *p++;
        path[i] = '\0';
        if (p < line_end) p++;

        /* version */
        i = 0;
        while (p < line_end && i < 15)
            version[i++] = *p++;
        version[i] = '\0';

        /* 提取 Host */
        char host[256];
        extract_host(text, text_len, host, sizeof(host));

        char url[2560];
        if (host[0])
            snprintf(url, sizeof(url), "http://%s%s", host, path);
        else
            snprintf(url, sizeof(url), "%s", path);

        printf("\n%s\n", SEPARATOR);
        printf("[%s] >>> HTTP 请求 <<<\n", ts);
        printf("  方法: %s  URL: %s  版本: %s\n", method, url, version);
        printf("  源: %s:%u -> 目的: %s:%u\n", src_ip, sport, dst_ip, dport);
        printf("  --- 请求头 ---\n");
        print_header_lines(text, line_end ? (int)(strstr(text, "\r\n\r\n") ? strstr(text, "\r\n\r\n") - text : text_len) : text_len);
        printf("%s\n\n", SEPARATOR);

    /* ===== HTTP 响应报文 ===== */
    } else if (starts_with(text, "HTTP/")) {
        const char *hdr_end = strstr(text, "\r\n\r\n");
        int header_len = hdr_end ? (int)(hdr_end - text) : text_len;
        const char *body = hdr_end ? hdr_end + 4 : NULL;
        int body_len = hdr_end ? text_len - header_len - 4 : 0;
        if (body_len < 0) body_len = 0;

        /* 状态行 */
        const char *status_end = strstr(text, "\r\n");
        int status_len = status_end ? (int)(status_end - text) : text_len;

        printf("\n%s\n", SEPARATOR);
        printf("[%s] <<< HTTP 响应 <<<\n", ts);
        printf("  状态行: %.*s\n", status_len, text);
        printf("  源: %s:%u -> 目的: %s:%u\n", src_ip, sport, dst_ip, dport);
        printf("  --- 响应头 ---\n");
        print_header_lines(text, header_len);

        if (body && body_len > 0) {
            printf("  --- 响应体 (%d bytes) ---\n", body_len);
            int print_len = body_len < MAX_PAYLOAD_PRINT ? body_len : MAX_PAYLOAD_PRINT;
            printf("    %.*s\n", print_len, body);
            if (body_len > MAX_PAYLOAD_PRINT)
                printf("    ... (truncated, total %d bytes)\n", body_len);
        }
        printf("%s\n\n", SEPARATOR);

    /* ===== HTTP 续传数据（响应体跨包） ===== */
    } else if (sport == 80 && payload_len > 0) {
        /* 检查是否有可打印内容 */
        int printable = 0;
        for (int j = 0; j < text_len && j < 100; j++) {
            if (text[j] >= 32 && text[j] < 127) {
                printable = 1;
                break;
            }
        }
        if (printable) {
            printf("[%s] [HTTP Data] %s:%u -> %s:%u (%d bytes)\n",
                   ts, src_ip, sport, dst_ip, dport, payload_len);
            int preview_len = text_len < 300 ? text_len : 300;
            printf("    %.*s\n", preview_len, text);
            if (text_len > 300)
                printf("    ... (truncated)\n");
        } else {
            printf("[%s] [HTTP Binary Data] (%d bytes)\n", ts, payload_len);
        }
    }
}

int main(int argc, char *argv[])
{
    char errbuf[PCAP_ERRBUF_SIZE];
    const char *iface = argc > 1 ? argv[1] : NULL;
    const char *bpf_filter = "(tcp port 80) or (ip6 and tcp port 80)";

    if (!iface) {
        pcap_if_t *alldevs;
        pcap_findalldevs(&alldevs, errbuf);
        static char default_dev[256];
        strncpy(default_dev, alldevs->d_name, sizeof(default_dev) - 1);
        iface = default_dev;
        pcap_freealldevs(alldevs);
    }

    printf("%s\n", SEPARATOR);
    printf("  实验1(第一部分) - HTTP 被动捕包 (libpcap C 版本)\n");
    printf("%s\n", SEPARATOR);
    printf("  接口: %s\n", iface);
    printf("  BPF 过滤器: %s\n", bpf_filter);

    char timestr[64];
    time_t now = time(NULL);
    strftime(timestr, sizeof(timestr), "%Y-%m-%d %H:%M:%S", localtime(&now));
    printf("  时间: %s\n", timestr);
    printf("  提示: 在另一个终端运行 curl http://today.hit.edu.cn/\n");
    printf("  按 Ctrl+C 停止捕获\n");
    printf("%s\n\n", SEPARATOR);

    pcap_t *handle = pcap_open_live(iface, SNAP_LEN, 1, 1000, errbuf);

    struct bpf_program fp;
    pcap_compile(handle, &fp, bpf_filter, 1, PCAP_NETMASK_UNKNOWN);
    pcap_setfilter(handle, &fp);
    pcap_freecode(&fp); 

    /* pcap_loop: 进入捕获循环，对每个包调用 packet_callback */
    pcap_loop(handle, -1 /* 无限捕获 */, packet_callback, NULL);

    pcap_close(handle);
    return 0;
}

libpcap 根据我们写的 bpf_filter 过滤器，自己把它包装进合法的 bpf 程序。

由于pcap是在协议栈后进行的，它拿到的包是系统本身收、发包的复制，同时具有一定的延迟。

part2

part2 直接用xdp在驱动层面进行过滤；如果舍弃包，协议栈后面的程序都拿不到这个包。

由于代码是用户态书写然后载入内核态的，eBPF的 verifier 会对 xdp.c 代码进行特别严格的检查，书写 xdp.c 会是一个难点。

// SPDX-License-Identifier: GPL-2.0
/*
 * 实验1(第二部分) - 基于 XDP/eBPF 的 HTTP 被动捕包
 *
 * 内核态 XDP 程序：在网卡驱动层捕获 TCP port 80 的数据包，
 * 提取 IP/TCP 头部信息和 HTTP 载荷，通过 perf event 发送到用户态。
 */

#include <linux/bpf.h>
#include <linux/if_ether.h>
#include <linux/in.h>
#include <linux/ip.h>
#include <linux/ipv6.h>
#include <linux/tcp.h>
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_endian.h>

#define MAX_PAYLOAD_SIZE 1024
#define HTTP_PORT        80

struct event {
    __u8  src_ip[16];   /* IPv4 映射为 ::ffff:x.x.x.x，IPv6 原样存储 */
    __u8  dst_ip[16];
    __u16 src_port;
    __u16 dst_port;
    __u8  tcp_flags;
    __u8  is_ipv6;
    __u8  _pad[2];
    __u32 payload_len;
    __u8  payload[MAX_PAYLOAD_SIZE];
};

/* perf event 输出 map，用于向用户态发送捕获事件 */
struct {
    __uint(type, BPF_MAP_TYPE_PERF_EVENT_ARRAY);
    __uint(key_size, sizeof(__u32));
    __uint(value_size, sizeof(__u32));
} events SEC(".maps");

/* per-CPU 临时缓冲区，避免栈溢出（eBPF 栈限制 512 字节） */
struct {
    __uint(type, BPF_MAP_TYPE_PERCPU_ARRAY);
    __uint(max_entries, 1);
    __type(key, __u32);
    __type(value, struct event);
} event_buf SEC(".maps");

SEC("xdp")
int xdp_http_capture(struct xdp_md *ctx)
{
    void *data     = (void *)(long)ctx->data;
    void *data_end = (void *)(long)ctx->data_end;

    /* ---------- 解析以太网头 ---------- */
    struct ethhdr *eth = data;
    if ((void *)(eth + 1) > data_end)
        return XDP_PASS;

    __u16 h_proto = eth->h_proto;
    struct tcphdr *tcp;
    __u32 ip_hdr_len;
    __u8  is_ipv6 = 0;

    /* 临时存储 IP 地址，后面填入 evt */
    __u8 tmp_src[16] = {};
    __u8 tmp_dst[16] = {};

    if (h_proto == bpf_htons(ETH_P_IP)) {
        /* ---------- IPv4 ---------- */
        struct iphdr *ip = (void *)(eth + 1);
        if ((void *)(ip + 1) > data_end)
            return XDP_PASS;
        if (ip->protocol != IPPROTO_TCP)
            return XDP_PASS;

        ip_hdr_len = ip->ihl * 4;
        if (ip_hdr_len < sizeof(struct iphdr) || ip_hdr_len > 60)
            return XDP_PASS;

        tcp = (void *)ip + ip_hdr_len;

        /* 存为 IPv4-mapped: 前 10 字节 0, 2 字节 0xff, 4 字节地址 */
        tmp_src[10] = 0xff; tmp_src[11] = 0xff;
        __builtin_memcpy(&tmp_src[12], &ip->saddr, 4);
        tmp_dst[10] = 0xff; tmp_dst[11] = 0xff;
        __builtin_memcpy(&tmp_dst[12], &ip->daddr, 4);

    } else if (h_proto == bpf_htons(ETH_P_IPV6)) {
        /* ---------- IPv6 ---------- */
        struct ipv6hdr *ip6 = (void *)(eth + 1);
        if ((void *)(ip6 + 1) > data_end)
            return XDP_PASS;
        if (ip6->nexthdr != IPPROTO_TCP)
            return XDP_PASS;

        ip_hdr_len = sizeof(struct ipv6hdr);  /* 固定 40 字节 */
        tcp = (void *)(ip6 + 1);
        is_ipv6 = 1;

        __builtin_memcpy(tmp_src, &ip6->saddr, 16);
        __builtin_memcpy(tmp_dst, &ip6->daddr, 16);
    } else {
        return XDP_PASS;
    }

    /* ---------- 解析 TCP 头 ---------- */
    if ((void *)(tcp + 1) > data_end)
        return XDP_PASS;

    __u16 src_port = bpf_ntohs(tcp->source);
    __u16 dst_port = bpf_ntohs(tcp->dest);

    /* 仅处理 HTTP 端口 */
    if (src_port != HTTP_PORT && dst_port != HTTP_PORT)
        return XDP_PASS;

    /* ---------- 填充事件结构体 ---------- */
    __u32 zero = 0;
    struct event *evt = bpf_map_lookup_elem(&event_buf, &zero);
    if (!evt)
        return XDP_PASS;

    __builtin_memcpy(evt->src_ip, tmp_src, 16);
    __builtin_memcpy(evt->dst_ip, tmp_dst, 16);
    evt->src_port  = src_port;
    evt->dst_port  = dst_port;
    evt->is_ipv6   = is_ipv6;
    evt->tcp_flags = ((__u8 *)tcp)[13];

    /* ---------- 提取 HTTP 载荷 ---------- */
    __u32 tcp_hdr_len = tcp->doff * 4;
    if (tcp_hdr_len < sizeof(struct tcphdr) || tcp_hdr_len > 60)
        return XDP_PASS;

    __u32 hdr_total = sizeof(struct ethhdr) + ip_hdr_len + tcp_hdr_len;
    __u32 pkt_len   = data_end - data;

    evt->payload_len = 0;
    /* 使用 __u64 避免编译器插入 32 位零扩展拷贝（<<32; >>32），
     * 使 asm volatile 屏障和 > 0 检查作用在同一个寄存器上，
     * 让 eBPF 验证器能正确收窄值域。 */
    __u64 payload_len = (__u64)((__u32)(pkt_len - hdr_total));
    asm volatile("" : "+r"(payload_len));
    if (payload_len > MAX_PAYLOAD_SIZE)
        payload_len = MAX_PAYLOAD_SIZE;
    if (payload_len > 0) {
        int ret = bpf_xdp_load_bytes(ctx, hdr_total, evt->payload, payload_len);
        if (ret == 0)
            evt->payload_len = (__u32)payload_len;
    }

    /* ---------- 发送事件到用户态 ---------- */
    __u64 size = __builtin_offsetof(struct event, payload) + evt->payload_len;
    if (size > sizeof(struct event))
        size = sizeof(struct event);

    bpf_perf_event_output(ctx, &events, BPF_F_CURRENT_CPU, evt, size);

    return XDP_PASS;
}

写对了后面就不难了，利用 go 的高并行特性写个前端代码，收到事件后就解析 payload 的内容。

go 语言中， := 类似于 cpp 中的 auto，自动推断变量类型，但写起来更简短；函数声明前缀 go func() 能够创建一个协程，这里就用以处理中断信号，及时关闭 perf event。

// 实验1(第二部分) - eBPF 用户态程序
// 加载 XDP 程序并从 perf event map 读取捕获的 HTTP 数据包
package main

import (
	"bytes"
	"encoding/binary"
	"fmt"
	"net"
	"os"
	"os/signal"
	"strings"
	"syscall"
	"time"

	"github.com/cilium/ebpf"
	"github.com/cilium/ebpf/link"
	"github.com/cilium/ebpf/perf"
)

// EventHeader 对应内核态 struct event 的头部（不含 payload）
type EventHeader struct {
	SrcIP      [16]byte
	DstIP      [16]byte
	SrcPort    uint16
	DstPort    uint16
	TCPFlags   uint8
	IsIPv6     uint8
	Pad        [2]byte
	PayloadLen uint32
}

const headerSize = 44 // 16+16+2+2+1+1+2+4

func addrToString(addr [16]byte, isV6 uint8) string {
	if isV6 == 0 {
		// IPv4-mapped: 最后 4 字节是 IPv4 地址
		return net.IP(addr[12:16]).String()
	}
	return net.IP(addr[:]).String()
}

func flagsToString(f uint8) string {
	names := []struct {
		mask uint8
		name string
	}{
		{0x01, "FIN"}, {0x02, "SYN"}, {0x04, "RST"},
		{0x08, "PSH"}, {0x10, "ACK"}, {0x20, "URG"},
	}
	var parts []string
	for _, n := range names {
		if f&n.mask != 0 {
			parts = append(parts, n.name)
		}
	}
	if len(parts) == 0 {
		return "NONE"
	}
	return strings.Join(parts, "|")
}

func main() {
	ifaceName := "enx6c1ff77293a3"
	fmt.Println("======================================")
	fmt.Println("  实验1(第二部分) - XDP HTTP 被动捕包")
	fmt.Println("======================================")
	fmt.Printf("  接口: %s\n", ifaceName)
	fmt.Printf("  时间: %s\n", time.Now().Format("2006-01-02 15:04:05"))
	fmt.Println("  提示: 在另一终端执行 curl http://today.hit.edu.cn/")
	fmt.Println("  按 Ctrl+C 停止")
	fmt.Println("======================================")
	fmt.Println()

	spec, _ := ebpf.LoadCollectionSpec("xdp.o")
	coll, _ := ebpf.NewCollection(spec)
	defer coll.Close()

	prog := coll.Programs["xdp_http_capture"]

	iface, _ := net.InterfaceByName(ifaceName)
	xdpLink, _ := link.AttachXDP(link.XDPOptions{
		Program:   prog,
		Interface: iface.Index,
	})
	defer xdpLink.Close()
	fmt.Printf("[+] XDP 程序已挂载到 %s\n\n", ifaceName)

	rd, _ := perf.NewReader(coll.Maps["events"], os.Getpagesize()*64)
	defer rd.Close()

	// 5. 信号处理，优雅退出
	sig := make(chan os.Signal, 1)
	signal.Notify(sig, syscall.SIGINT, syscall.SIGTERM)
	go func() {
		<-sig
		fmt.Println("\n[*] 正在卸载 XDP 程序...")
		rd.Close()
	}()

	for {
		record, err := rd.Read()
		if err != nil {
			break
		}

		raw := record.RawSample
		var hdr EventHeader
		binary.Read(bytes.NewReader(raw[:headerSize]), binary.LittleEndian, &hdr)

		srcIP := addrToString(hdr.SrcIP, hdr.IsIPv6)
		dstIP := addrToString(hdr.DstIP, hdr.IsIPv6)
		flags := flagsToString(hdr.TCPFlags)
		ts := time.Now().Format("15:04:05")

		fmt.Printf("[%s] [%s] %s:%d -> %s:%d", ts, flags, srcIP, hdr.SrcPort, dstIP, hdr.DstPort)

		if hdr.PayloadLen > 0 && len(raw) >= headerSize+int(hdr.PayloadLen) {
			payload := string(raw[headerSize : headerSize+int(hdr.PayloadLen)])

			if strings.HasPrefix(payload, "GET ") ||
				strings.HasPrefix(payload, "POST ") ||
				strings.HasPrefix(payload, "HEAD ") {
				lines := strings.SplitN(payload, "\r\n", -1)
				fmt.Printf("\n  >>> HTTP 请求: %s\n", lines[0])
				for _, hl := range lines[1:] {
					if hl == "" {
						break
					}
					fmt.Printf("      %s\n", hl)
				}
			} else if strings.HasPrefix(payload, "HTTP/") {
				parts := strings.SplitN(payload, "\r\n\r\n", 2)
				headerLines := strings.Split(parts[0], "\r\n")
				fmt.Printf("\n  <<< HTTP 响应: %s\n", headerLines[0])
				for _, hl := range headerLines[1:] {
					fmt.Printf("      %s\n", hl)
				}
				if len(parts) > 1 && len(parts[1]) > 0 {
					body := parts[1]
					if len(body) > 500 {
						body = body[:500] + "..."
					}
					fmt.Printf("  Body:\n      %s\n", body)
				}
			} else {
				fmt.Printf(" [%d bytes payload]", hdr.PayloadLen)
			}
		}
		fmt.Println()
	}
}
