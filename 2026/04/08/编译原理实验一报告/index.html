编译原理实验一报告

整个实验的任务是要实现一个特色的C–语言的词法分析和语法分析。

可以根据附件的 C-- 语言文法示意图，定义 token 和文法规则。

Tree部分

因为实验要求打印语法树，定义了结构体和一些建树的函数

tree.h：

#ifndef TREE_H
#define TREE_H

#include <stdarg.h>

/*
因为当过 oi/acm 选手，我更倾向于用抽象的数组而不是结构体来表示树/节点。
但为了代码的易维护性，这里就学着用了结构体。
*/

typedef struct Node {
    char *name;
    int lineno;
    int is_token;
    char *extra;
    char *token_name;
    struct Node *child;
    struct Node *sibling;
} Node;

Node *make_node(const char *name, int lineno, int argc, ...);
Node *make_token(const char *name, int lineno, const char *extra, const char *token_name);
void print_tree(Node *root, int indent);
void print_stmt_ast(Node *stmt);

extern int has_error;
extern int analyze_flag;

#endif

tree.c：

#include "tree.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int has_error = 0;
int analyze_flag = 0;

/*
语法树并不总是二叉树。
这个函数的目标是将多个子节点连接到"name"这个节点上，因此使用了可变参数和 va_list。
父节点一般是非终结符，是需要我们创造的；bison 调用这个函数时，参数应该都是从栈中拿出来的，$1, $2, $3......
*/

Node *make_node(const char *name, int lineno, int argc, ...) {
    Node *node = (Node *)calloc(1, sizeof(Node));
    node->name = strdup(name);
    node->lineno = lineno;
    va_list ap;
    va_start(ap, argc);
    Node *last = NULL;
    for (int i = 0; i < argc; i++) {
        Node *c = va_arg(ap, Node *);
        if (!c) continue;
        if (!node->child) {
            node->child = c;
            node->lineno = c->lineno;
        } else {
            /*
            所谓 sibling 其实就是 next 指针，父节点要遍历儿子时就是先 ->child 然后不断 ->sibling，直到 NULL。
            father 0
                   |
            son    1-sibling->2-sibling->3-sibling-> NULL
            */
            last->sibling = c;
        }
        last = c;
    }
    va_end(ap);
    return node;
}

/*
这个函数用于创建一个 token 节点，它们毫无疑问是语法树的叶子节点。
*/

Node *make_token(const char *name, int lineno, const char *extra, const char *token_name) {
    Node *node = (Node *)calloc(1, sizeof(Node));
    node->name = strdup(name);
    node->lineno = lineno;
/*
is_token 这个参数看起来有点没有必要，事实上叶子节点一定是 token。
不过要判断是否是叶子节点需要看它是否有子节点，直接建一个参数大概更直观一点。
*/
    node->is_token = 1;
    node->extra = extra ? strdup(extra) : NULL;
    node->token_name = token_name ? strdup(token_name) : NULL;
    return node;
}


/*
用树枝字符打印 AST。
*/

static void print_node_label(Node *node) {
    if (node->is_token) {
        printf("%s", node->name);
        if (node->token_name) {
            printf(" %s", node->token_name);
        } else if (node->extra) {
            printf(" %s", node->extra);
            // printf(" %s", node->extra);
        }
        printf("\n");
    } else {
        printf("%s (%d)\n", node->name, node->lineno);
    }
}

static int needs_space_before(const char *prev_name, const char *curr_name, const char *curr_text) {
    (void)curr_name;

    if (!prev_name) return 0;
    if (!curr_text) return 1;

    if (!strcmp(curr_text, ";") || !strcmp(curr_text, ",") || !strcmp(curr_text, ")") ||
        !strcmp(curr_text, "]") || !strcmp(curr_text, "}")) {
        return 0;
    }

    if (!strcmp(curr_text, "[")) return 0;
    /*除非该"（"是在函数的声明中使用，否则意味这它在表达式中，要在前面加上一个空格*/
    if (!strcmp(curr_text, "(")) return strcmp(prev_name, "ID") != 0;
    if (!strcmp(prev_name, "LB") || !strcmp(prev_name, "LP") || !strcmp(prev_name, "LC")) return 0;

    return 1;
}

static void print_stmt_source(Node *node, const char **prev_name) {
    if (!node) return;

    if (node->is_token) {
        const char *text = node->token_name ? node->token_name : node->extra;
        if (text) {
            if (needs_space_before(*prev_name, node->name, text)) putchar(' ');
            fputs(text, stdout);
            *prev_name = node->name;
        }
        return;
    }

    for (Node *c = node->child; c; c = c->sibling) {
        print_stmt_source(c, prev_name);
    }
}

static Node *next_visible_sibling(Node *node) {
    for (Node *c = node->sibling; c; c = c->sibling) {
        if (c->is_token || c->child) return c;
    }
    return NULL;
}

static void print_tree_branch(Node *node, const char *prefix, int is_last) {
    char next_prefix[1024];

    if (!node) return;
    if (!node->is_token && !node->child) return;

    // printf("%s%s", prefix, is_last ? "└── " : "├── ");
    printf("%s%s", prefix, is_last ? "  " : "  ");
    print_node_label(node);

    // snprintf(next_prefix, sizeof(next_prefix), "%s%s", prefix, is_last ? "    " : "│   ");
    snprintf(next_prefix, sizeof(next_prefix), "%s%s", prefix, is_last ? "  " : "  ");
    for (Node *c = node->child; c; c = c->sibling) {
        if (!c->is_token && !c->child) continue;
        print_tree_branch(c, next_prefix, next_visible_sibling(c) == NULL);
    }
}

void print_tree(Node *root, int indent) {
    (void)indent;

    if (!root) return;
    if (!root->is_token && !root->child) return;

    print_node_label(root);
    for (Node *c = root->child; c; c = c->sibling) {
        if (!c->is_token && !c->child) continue;
        print_tree_branch(c, "", next_visible_sibling(c) == NULL);
    }
}

void print_stmt_ast(Node *stmt) {
    if (!stmt || !analyze_flag) return;
    const char *prev_name = NULL;
    printf("Statement: ");
    print_stmt_source(stmt, &prev_name);
    putchar('\n');
    printf("Statement AST:\n");
    print_tree(stmt, 0);
    putchar('\n');
}

bison部分

先看bison的部分，首先是 token，也就是终结符：

INT FLOAT ID Type
SEMI COMMA ASSIGNOP RELOP
PLUS MINUS STAR DIV
ND OR NOT DOT
LP RP LB RB LC RC
STRUCT RETURN IF ELSE WHILE

然后是非终结符：

/* non-terminals */
%type <node> Program ExtDefList ExtDef ExtDecList
%type <node> Specifier StructSpecifier OptTag Tag
%type <node> VarDec FunDec VarList ParamDec
%type <node> CompSt StmtList Stmt
%type <node> DefList Def DecList Dec
%type <node> Exp Args

同时，对于以下终结符（和运算有关），需要确定它们的优先级：

%left 表示左结合，%right 表示右结合，%nonassoc 表示不结合，优先级从低到高。

/* precedence (low → high) */
%nonassoc LOWER_THAN_ELSE //
%nonassoc ELSE // else的优先级最低
%right ASSIGNOP // 赋值运算右结合，从右到左进行，即 a=b=c <=> a=(b=c)
%left OR // 左结合
%left AND // 定义了AND的优先级高于OR
%left RELOP // >= <= 等关系运算符
%left PLUS MINUS // + -
%left STAR DIV // 乘除高于加减
%right NOT // !的优先级更高
%left LP RP LB RB DOT // ( ) [ ] . 的优先级最高

其中，LOWER_THAN_ELSE 定义一个虚拟的最低优先级，ELSE 需要比”没有 else 的 if” 优先级更高。

通常语言都规定：else 和最近的那个还没匹配的 if 结合。为了避免发生移进/归约冲突，我们就让 ELSE 的优先级高于 LOWER_THAN_ELSE。

  if a then  // else 的优先级需要高于这个if
      if b then // 这样它才会匹配这个 if
          s1
      else
          s2

flex 的任务就是正则匹配，匹配到后执行 make_token 的动作，生成叶子节点。
