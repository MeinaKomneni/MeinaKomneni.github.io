#!/usr/bin/env python3
import argparse
from collections import deque
from pathlib import Path
from urllib.parse import urljoin, urlparse

import requests
from bs4 import BeautifulSoup

RAW_HEADERS = '''
User-Agent: Mozilla/5.0 (X11; Ubuntu; Linux x86_64; rv:149.0) Gecko/20100101 Firefox/149.0
Accept: text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8
Accept-Language: zh-CN,zh;q=0.9,zh-TW;q=0.8,zh-HK;q=0.7,en-US;q=0.6,en;q=0.5
Accept-Encoding: gzip, deflate, br
Upgrade-Insecure-Requests: 1
Sec-Fetch-Dest: document
Sec-Fetch-Mode: navigate
Sec-Fetch-Site: none
Sec-Fetch-User: ?1
Priority: u=0, i
Te: trailers
'''

HEADERS = dict(
    line.split(": ", 1)
    for line in RAW_HEADERS.strip().splitlines()
)

def parse_args():
    parser = argparse.ArgumentParser(description="新闻网页爬虫（支持 BFS / DFS）")
    parser.add_argument("url", help="起始网页 URL")
    parser.add_argument(
        "--mode",
        choices=["bfs", "dfs"],
        default="bfs",
        help="爬取策略：bfs 或 dfs",
    )
    parser.add_argument(
        "--max-pages",
        type=int,
        default=10,
        help="最大爬取页面数量",
    )
    parser.add_argument(
        "--max-depth",
        type=int,
        default=2,
        help="最大爬取深度",
    )
    parser.add_argument(
        "--output",
        default=None,
        help="输出文件路径，默认按模式生成 txt 文件",
    )
    return parser.parse_args()

def fetch_page(url):
    response = requests.get(
        url,
        headers=HEADERS,
        timeout=10,
    )
    response.raise_for_status()
    response.encoding = response.apparent_encoding or response.encoding
    # print(response.text)
    return response.text

'''
如果 mode 是 bfs，那就使用队列；如果是 dfs 则使用栈（而不是函数递归调用）。
'''

def crawl(start_url, mode, max_pages, max_depth):
    visited = set()
    results = []
    if mode == "bfs":
        frontier = deque([(start_url, 0)])
    else:
        frontier = [(start_url, 0)]

    while frontier and len(results) < max_pages:
        if mode == "bfs":
            url, depth = frontier.popleft()
        else:
            url, depth = frontier.pop()

        if depth > max_depth or url in visited:
            continue

        visited.add(url)
        print(f"[{mode.upper()}] 正在爬取: depth={depth} url={url}")
        html = fetch_page(url)

        soup = BeautifulSoup(html, "html.parser")
        # 提取标题
        title = soup.title.get_text(strip=True)
        # 提取正文
        content = "\n".join(
            p.get_text(" ", strip=True)
            for p in soup.find_all("p")
            if p.get_text(" ", strip=True)
        )
        # 提取 links
        links = [
            urljoin(url, tag["href"])
            for tag in soup.find_all("a", class_="links", href=True)
        ]

        results.append(
            {
                "url": url,
                "depth": depth,
                "title": title,
                "content": content,
                "links": links,
            }
        )

        if depth == max_depth:
            continue

        next_items = []
        for link in links:
            if link not in visited:
                next_items.append((link, depth + 1))

        if mode == "bfs":
            frontier.extend(next_items)
        else:
            frontier.extend(reversed(next_items))

    return results

def save_results(results, output_path, mode):
    output = Path(output_path)
    with output.open("w", encoding="utf-8") as file:
        file.write(f"爬取模式: {mode.upper()}\n")
        file.write(f"共保存 {len(results)} 个页面\n")
        file.write("=" * 80 + "\n\n")

        for index, item in enumerate(results, start=1):
            file.write(f"第 {index} 篇\n")
            file.write(f"URL: {item['url']}\n")
            file.write(f"深度: {item['depth']}\n")
            file.write(f"标题: {item['title']}\n")
            file.write("正文:\n")
            file.write(item["content"] + "\n\n")
            file.write("推荐链接:\n")
            if item["links"]:
                for link in item["links"][:20]:
                    file.write(f"- {link}\n")
            else:
                file.write("- 无\n")
            file.write("\n" + "-" * 80 + "\n\n")

def main():
    args = parse_args()
    output_path = args.output or f"{args.mode}_result.txt"
    results = crawl(args.url, args.mode, args.max_pages, args.max_depth)
    save_results(results, output_path, args.mode)
    print(f"爬取完成，共保存 {len(results)} 个页面到 {output_path}")

if __name__ == "__main__":
    main()
