from typing import Union

import numpy as np
from scipy import sparse

ArrayLike = Union[np.ndarray, sparse.spmatrix]

class MultinomialNaiveBayes:
    def __init__(self, alpha: float = 1.0):
        self.alpha = alpha
        self.classes_: np.ndarray  = None
        self.class_log_prior_: np.ndarray = None
        self.feature_log_prob_: np.ndarray = None

    def fit(self, X: ArrayLike, y: np.ndarray) -> "MultinomialNaiveBayes":
        '''
        X 是 train_data, Y 是 train_tags。
        '''
        if sparse.issparse(X): # 判断 X 是不是稀疏矩阵
            X = X.tocsr() # 把稀疏矩阵转成 CSR（Compressed Sparse Row） 格式，即按行压缩存储。

            '''
            CSR 用三个数组来存储：
            - data — 所有非零值                                                                                                                                   
            - indices — 每个非零值的列索引                                                                                                                        
            - indptr — 每一行的非零元素在 data 中从哪开始
            举个例子，假设原矩阵是：                                                                                                                                
                列0  列1  列2  列3  列4                              
            行0 [ 0,   3,   0,   0,   2 ]                                                                                                                         
            行1 [ 0,   0,   1,   0,   0 ]                                                                                                                         
            行2 [ 4,   0,   0,   5,   0 ]                                                                                                                         
                                                                                                                                             
            CSR 存储为：                                                                                                                                                
            data    = [3, 2, 1, 4, 5]       # 所有非零值              
            indices = [1, 4, 2, 0, 3]       # 对应的列号                                                                                                          
            indptr  = [0, 2, 3, 5]          # 行0从data[0]开始，行1从data[2]开始，行2从data[3]开始
            最后的 5 是哨兵值（sentinel），表示最后一行的结束位置。 

            从使用角度看，X 还是一个矩阵对象，仍然可以正常做 X[y == c]、X.shape、X.sum() 这些操作，和普通矩阵一样。
            '''
        else:
            X = np.array(X)

        self.classes_ = np.unique(y) # 从训练标签中提取出所有类别
        n_samples = X.shape[0] # 训练样本数
        n_features = X.shape[1] # 特征数
        n_classes = len(self.classes_) # 类别数

        self.class_log_prior_ = np.zeros(n_classes)
        self.feature_log_prob_ = np.zeros((n_classes, n_features))

        for i, c in enumerate(self.classes_):
            X_c = X[y == c] # 提取所有 tag 为 c 的矩阵
            # 先验概率 P(c) = 该类别文档数 / 总文档数，然后取 log，这样乘法就可以变成 log 的加法。
            self.class_log_prior_[i] = np.log(X_c.shape[0] / n_samples)

            # 每个特征在该类别中的总次数
            if sparse.issparse(X_c):
                '''
                X_c 是类别 c 的所有文档组成的矩阵，形状比如 (1600, 30000)，每行是一篇文档的词频向量。                                                            
                X_c.sum(axis=0) 就是按列求和——把同一列（同一个词）在所有文档中的词频加起来。
                文档1: [0, 3, 0, 1, ...]                                                                                                                              
                文档2: [0, 1, 0, 2, ...]                                                                                                                              
                文档3: [0, 0, 0, 5, ...]
                        ↓  ↓  ↓  ↓   
                求和:  [0, 4, 0, 8, ...] 
                结果 feature_count 是一个长度 30000 的向量，第 j 个位置就是词 j 在类别 c 的所有文档中一共出现了多少次。                                                         
                .flatten() 是因为稀疏矩阵 sum(axis=0) 返回的是 (1, 30000) 的矩阵，需要压平成一维数组。
                '''
                feature_count = np.asarray(X_c.sum(axis=0)).flatten()
            else:
                feature_count = X_c.sum(axis=0)

            # 拉普拉斯平滑 + 取对数
            total_count = feature_count.sum()
            self.feature_log_prob_[i] = np.log(
                (feature_count + self.alpha) / (total_count + self.alpha * n_features)
            ) # 计算 P(特征/类别)
        # 最终 feature_log_prob_[i] 是一个长度 30000 的向量，存的是类别 i 下每个词的 log P(w_j | c)。
        return self

    def predict(self, X: ArrayLike) -> np.ndarray:
        if sparse.issparse(X):
            X = X.tocsr()
        # 对数后验 = X · log_prob^T + log_prior
        '''
        X:                (n篇文档, 30000)    — 每行是一篇文档的词频                                                                                          
        feature_log_prob: (5个类别, 30000)    — 每行是一个类别的 log P(w|c) 
        X · feature_log_prob^T → (n篇文档, 5)  — 每个格子就是某篇文档对某个类别的词频加权得分 log P(w_j | c)
        + class_log_prior_     → (n篇文档, 5)  — 再加上先验 P(c) = 该类别文档数 / 总文档数
        '''

        log_posterior = X.dot(self.feature_log_prob_.T) + self.class_log_prior_
        if sparse.issparse(log_posterior):
            log_posterior = log_posterior.toarray()
        indices = np.argmax(log_posterior, axis=1)
        return self.classes_[indices]

推导

根据贝叶斯定理，给定文档 d 和类别 c，其后验概率为：

P(c∣d)=P(d∣c)P(c)P(d)P(c|d) = \frac{P(d|c)P(c)}{P(d)} P(c∣d)=P(d)P(d∣c)P(c)​

由于分母 P(d) 对所有类别均相同，分类时只需比较分子大小：

P(c∣d)∝P(d∣c)P(c)P(c|d) \propto P(d|c)P(c) P(c∣d)∝P(d∣c)P(c)

类别 c 的先验概率由训练集中该类别的文档占比估计：

P(c)=NcNP(c) = \frac{N_c}{N} P(c)=NNc​​

其中 N_c 为类别 c 的文档数，N 为总文档数。

设文档 d 由词 w_1, w_2, ..., w_n 组成，多项式朴素贝叶斯模型假设各词条件独立，从而有：

P(d∣c)=P(w1,w2,...,wn∣c)=∏j=1∣V∣P(wj∣c)njP(d|c) = P(w_1, w_2, ..., w_n|c) = \prod_{j=1}^{|V|} P(w_j|c)^{n_j} P(d∣c)=P(w1​,w2​,...,wn​∣c)=j=1∏∣V∣​P(wj​∣c)nj​

其中：

  • |V| 为词汇表大小（即特征维度）
  • n_j 为词 w_j 在文档 d 中出现的次数

最终，取后验概率最大的类别作为文档 d 的预测结果：

c^=arg⁡max⁡c∈CP(c∣d)\hat{c} = \arg\max_{c \in C} P(c|d) c^=argc∈Cmax​P(c∣d)

平滑与数值稳定性

为避免词语从未出现在某类别中导致的零概率问题，采用拉普拉斯平滑来估计条件概率：

P(wj∣c)=Njc+α∑k=1∣V∣Nkc+α∣V∣P(w_j|c) = \frac{N_{jc} + \alpha}{\sum_{k=1}^{|V|}N_{kc} + \alpha|V|} P(wj​∣c)=∑k=1∣V∣​Nkc​+α∣V∣Njc​+α​

其中：

  • N_jc 为词 w_j 在类别 c 的所有文档中出现的总次数
  • α 为平滑参数（通常取 1）

对应的代码实现为：

(feature_count + self.alpha) / (total_count + self.alpha * n_features)

为防止数值下溢并将连乘转化为求和，所有概率均取对数后在对数空间中计算：

log⁡P(c∣d)=log⁡P(c)+∑j=1∣V∣njlog⁡P(wj∣c)\log P(c|d) = \log P(c) + \sum_{j=1}^{|V|} n_j \log P(w_j|c) logP(c∣d)=logP(c)+j=1∑∣V∣​nj​logP(wj​∣c)

在代码中，上式对应如下矩阵运算：

log_posterior=X⋅(feature_log_prob)T+class_log_prior\text{log\_posterior} = X \cdot (\text{feature\_log\_prob})^T + \text{class\_log\_prior} log_posterior=X⋅(feature_log_prob)T+class_log_prior

其中 X 为文档-词频矩阵，元素 X[i,j] = n_j 表示词 j 在文档 i 中的出现次数。

最终通过取对数后验概率最大的类别完成分类：

c^=arg⁡max⁡clog⁡P(c∣d)\hat{c} = \arg\max_c \log P(c|d) c^=argcmax​logP(c∣d)

在代码中由 np.argmax(log_posterior, axis=1) 实现。
