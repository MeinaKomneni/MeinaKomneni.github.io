AC算法的实现

Trie 树 + fail 函数 = AC 自动机

AC 自动机本质上是 Trie 上的自动机。可以先按 Trie 字典树的插入方式插入模式串。

这里应实验的要求，使用了动态分配的方法。

#include <bits/stdc++.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/resource.h>
#include <ext/pb_ds/assoc_container.hpp>
#include <ext/pb_ds/tree_policy.hpp>

using namespace std;

namespace AC {

struct Tire_Node {
    Tire_Node* fail_node;
    Tire_Node* next_node[256];
    int cnt = 0;
};

vector<Tire_Node*> pool;

Tire_Node* get_node() {
    Tire_Node* node = (Tire_Node*)malloc(sizeof(Tire_Node));
    memset(node, 0, sizeof(Tire_Node));
    pool.push_back(node);
    return node;
}

Tire_Node* root = get_node();

void reset() {
    for (auto p : pool) free(p);
    pool.clear();
    root = get_node();
}

// AC 自动机本质上是 Trie 上的自动机。可以先按 Trie 字典树的插入方式插入模式串。

void insert(const string &s) {
    Tire_Node* u = root;
    for (unsigned char c : s) {
        if (!u->next_node[c]) u->next_node[c] = get_node();
        u = u->next_node[c];
    }
    u->cnt++;
}

关键要学习的地方在于 fail 函数的构建。

void build() {
    queue<Tire_Node*> q;
    root->fail_node = root;
    for (int c = 0; c < 256; c++) {
        if (root->next_node[c]) {
            root->next_node[c]->fail_node = root;
            q.push(root->next_node[c]);
        } else {
            root->next_node[c] = root; // 路径压缩：空转移指回 root
        }
    }
    while (!q.empty()) {
        Tire_Node* u = q.front();
        q.pop();
        for (int c = 0; c < 256; c++) {
            Tire_Node* v = u->next_node[c];
            if (v) {
                v->fail_node = u->fail_node->next_node[c];
                v->cnt += v->fail_node->cnt;
                q.push(v);
            } else {
                u->next_node[c] = u->fail_node->next_node[c]; // 路径压缩
            }
        }
    }
}

Trie 上每个节点看作一个状态 uuu。

  • 记 next(u,c)\mathrm{next}(u,c)next(u,c) 为沿字符 c∈Σc\in\Sigmac∈Σ 的“显式” Trie 边（若不存在则为空）。
  • 记 go(u,c)\mathrm{go}(u,c)go(u,c) 为 AC 自动机的转移函数（含失配跳转后的转移）。
  • 记 fail(u)\mathrm{fail}(u)fail(u) 为失配指针。

1) fail 的初始化（根与第一层）

根状态 rrr 的失配指针指向自身：

fail(r)=r.\mathrm{fail}(r)=r. fail(r)=r.

对根的每个字符转移，若存在子节点 v=next(r,c)v={next}(r,c)v=next(r,c)，则

fail(v)=r.\mathrm{fail}(v)=r. fail(v)=r.

若根没有该字符的显式边，为了让转移函数总是有定义，令

go(r,c)=r.\mathrm{go}(r,c)=r. go(r,c)=r.

这对应代码里把 root->next_node[c] 为空的情况直接指回 root（路径压缩）。

2) 一般状态的 fail 构建

按 BFS（从浅到深）处理每个状态 uuu，对每个字符 ccc：

  • 若存在子节点 v=next(u,c)v={next}(u,c)v=next(u,c)，则其失配指针为

fail(v)=go(fail(u),c).\mathrm{fail}(v)=\mathrm{go}(\mathrm{fail}(u),c). fail(v)=go(fail(u),c).

同时把输出信息沿失配累加（对应代码的 cnt）：

out(v)  ←  out(v)+out(fail(v)).\mathrm{out}(v)\;\leftarrow\;\mathrm{out}(v)+\mathrm{out}(\mathrm{fail}(v)). out(v)←out(v)+out(fail(v)).

  • 若不存在显式边，则自动机在该字符上的转移应当等价于沿失配继续走：

go(u,c)=go(fail(u),c).\mathrm{go}(u,c)=\mathrm{go}(\mathrm{fail}(u),c). go(u,c)=go(fail(u),c).

这对应代码里的：u->next_node[c] = u->fail_node->next_node[c]。

3) 状态转移函数

因此在构建完成后，转移可以统一写为

δ(u,c)=go(u,c).\delta(u,c)=\mathrm{go}(u,c). δ(u,c)=go(u,c).

匹配文本时维护当前状态 uuu，读入字符 ccc 后直接做

u←δ(u,c),u \leftarrow \delta(u,c), u←δ(u,c),

并在到达状态 uuu 时累加其输出（例如统计匹配次数时加上 out(u){out}(u)out(u)）。

WM 算法的实现

WM 算法理解起来相对比较简单。当 B=2B = 2B=2 时，可以直接开一个大数组[65536]来实现完美哈希；但这里代码为了兼容其他数值的 BBB，使用了 unordered_map。


#include <bits/stdc++.h>
#include <unistd.h>
#include <fcntl.h>
#include <sys/resource.h>
#include <ext/pb_ds/assoc_container.hpp>
#include <ext/pb_ds/tree_policy.hpp>

using namespace std;

namespace WM {

const long long B = 2;
const long long INF = (1LL << 60);

long long n, m;
vector<pair<string, long long>> pat;
long long default_shift;
unordered_map<string, long long> shift;
unordered_map<string, vector<long long>> bucket;
long long cnt1[256];

void reset() {
    n = 0;
    pat.clear();
    shift.clear();
    bucket.clear();
    memset(cnt1, 0, sizeof(cnt1));
}

void build() {
    m = INF;
    for (int i = 0; i < n; i++) {
        pat[i].second = pat[i].first.size();
        m = min(m, pat[i].second);
    }

    default_shift = max(0LL, m - B + 1);

    for (int i = 0; i < n; i++) {
        if (pat[i].second < B) {
            if (pat[i].second == 1)
                cnt1[(unsigned char)pat[i].first[0]]++;
            continue;
        }
        for (long long j = 0; j <= m - B; j++) {
            string key = pat[i].first.substr(j, B);
            auto it = shift.find(key);
            long long val = m - B - j;
            if (it == shift.end())
                shift[key] = min(default_shift, val);
            else
                it->second = min(it->second, val);
        }
        string bkey = pat[i].first.substr(m - B, B);
        bucket[bkey].push_back(i);
    }
}

int query(const string &text) {
    int res = 0;
    int L = text.size();

    for (char ch : text) res += cnt1[(unsigned char)ch];

    if (m < B || L < (long long)m) return res;

    long long i = m - 1;
    while (i < L) {
        string key = text.substr(i - B + 1, B);
        auto it = shift.find(key);
        long long s = (it != shift.end()) ? it->second : default_shift;
        if (s) {
            i += s;
            continue;
        }
        auto bit = bucket.find(key);
        if (bit != bucket.end()) {
            for (int id : bit->second) {
                long long st = i - m + 1;
                if (st < 0 || st + pat[id].second > L) continue;
                bool ok = 1;
                for (long long j = 0; j < pat[id].second; j++) {
                    if (text[st + j] != pat[id].first[j]) {
                        ok = 0;
                        break;
                    }
                }
                res += ok;
            }
        }
        i++;
    }
    return res;
}

} // namespace WM

#ifndef MIX_MODE
using namespace WM;
long long get_memory_kb() {
    struct rusage usage;
    getrusage(RUSAGE_SELF, &usage);
    return usage.ru_maxrss; // KB on Linux
}

int main(int argc, char* argv[]) {
    cin.tie(0)->sync_with_stdio(0);

    cerr << "初始内存使用情况: " << get_memory_kb() << " KB\n";

    int fd = open(argv[1], O_RDONLY);
    dup2(fd, STDIN_FILENO);
    close(fd);

    string s;
    while (cin >> s) pat.emplace_back(s, 0);
    n = pat.size();

    auto t0 = chrono::high_resolution_clock::now();
    build();
    auto t1 = chrono::high_resolution_clock::now();

    cerr << "预处理时间花费: "
         << chrono::duration<double, milli>(t1 - t0).count() << " ms\n";
    cerr << "预处理后内存使用情况: " << get_memory_kb() << " KB\n";

    fd = open(argv[2], O_RDONLY);
    dup2(fd, STDIN_FILENO);
    close(fd);
    cin.clear();
    string text;

    auto t2 = chrono::high_resolution_clock::now();
    while (cin >> text) query(text);
    auto t3 = chrono::high_resolution_clock::now();

    cerr << "匹配时间花费: "
         << chrono::duration<double, milli>(t3 - t2).count() << " ms\n";
    cerr << "匹配后内存使用情况: " << get_memory_kb() << " KB\n";

    return 0;
}

AC+WM 并行

关键在与对字符串的分配。

设模式集 P={p1,p2,…,pN}\mathcal{P} = \{p_1, p_2, \ldots, p_N\}P={p1​,p2​,…,pN​} 按长度升序排列，即 ∣p1∣≤∣p2∣≤⋯≤∣pN∣|p_1| \leq |p_2| \leq \cdots \leq |p_N|∣p1​∣≤∣p2​∣≤⋯≤∣pN​∣。文本集 T={t1,t2,…,tM}\mathcal{T} = \{t_1, t_2, \ldots, t_M\}T={t1​,t2​,…,tM​}，总文本字符数 L=∑j∣tj∣L = \sum_j |t_j|L=∑j​∣tj​∣。

将 P\mathcal{P}P 划分为两个不相交的子集：

P=PAC∪PWM,PAC∩PWM=∅\mathcal{P} = \mathcal{P}_{\text{AC}} \cup \mathcal{P}_{\text{WM}}, \quad \mathcal{P}_{\text{AC}} \cap \mathcal{P}_{\text{WM}} = \emptyset P=PAC​∪PWM​,PAC​∩PWM​=∅

两个阶段均使用双线程并行：

Ttotal=Tprep+Tmatch=max⁡ ⁣(TACprep,  TWMprep)+max⁡ ⁣(TACmatch,  TWMmatch)T_{\text{total}} = T_{\text{prep}} + T_{\text{match}} = \max\!\left(T_{\text{AC}}^{\text{prep}},\; T_{\text{WM}}^{\text{prep}}\right) + \max\!\left(T_{\text{AC}}^{\text{match}},\; T_{\text{WM}}^{\text{match}}\right) Ttotal​=Tprep​+Tmatch​=max(TACprep​,TWMprep​)+max(TACmatch​,TWMmatch​)

目标：选择划分使 TtotalT_{\text{total}}Ttotal​ 最小。

AC 自动机的 build 阶段对每个 Trie 节点遍历全部 ∣Σ∣=256|\Sigma| = 256∣Σ∣=256 个转移来压缩失配链接。设 AC 侧模式串总字符数为 SACS_{\text{AC}}SAC​，Trie 节点数 ≈SAC\approx S_{\text{AC}}≈SAC​，则：

TACprep=γ⋅∣Σ∣⋅SACT_{\text{AC}}^{\text{prep}} = \gamma \cdot |\Sigma| \cdot S_{\text{AC}} TACprep​=γ⋅∣Σ∣⋅SAC​

其中 γ\gammaγ 为单次数组写入的代价常数。由于 TACprepT_{\text{AC}}^{\text{prep}}TACprep​ 与 SACS_{\text{AC}}SAC​ 成正比，且因为 ∣Σ∣=256|\Sigma| = 256∣Σ∣=256 的乘数使其成为主要的预处理瓶颈，查询时每个文本字符恰好触发一次数组索引跳转：

TACmatch=α⋅LT_{\text{AC}}^{\text{match}} = \alpha \cdot L TACmatch​=α⋅L

WM build 阶段对每个模式串遍历前 mmin⁡m_{\min}mmin​ 个位置进行 hash 插入。设 hash 操作相对数组访问的代价倍数为 ChC_hCh​，则：

TWMprep=Ch⋅nWM⋅mmin⁡T_{\text{WM}}^{\text{prep}} = C_h \cdot n_{\text{WM}} \cdot m_{\min} TWMprep​=Ch​⋅nWM​⋅mmin​

由于 ChC_hCh​ 的量级（≈40\approx 40≈40）远小于 ∣Σ∣=256|\Sigma| = 256∣Σ∣=256，且 nWM⋅mmin⁡≤SWMn_{\text{WM}} \cdot m_{\min} \leq S_{\text{WM}}nWM​⋅mmin​≤SWM​，WM 预处理通常远轻于 AC；WM 基于长度为 BBB 的 BBB-gram shift 表进行跳跃。设 PWM\mathcal{P}_{\text{WM}}PWM​ 中最短模式串长度为 mmin⁡m_{\min}mmin​，则每步最多跳过 d=mmin⁡−B+1d = m_{\min} - B + 1d=mmin​−B+1 个字符。每步的代价来自一次哈希表查找，因此：

TWMmatch=Ch⋅LdeffT_{\text{WM}}^{\text{match}} = \frac{C_h \cdot L}{d_{\text{eff}}} TWMmatch​=deff​Ch​⋅L​

其中 deffd_{\text{eff}}deff​ 为考虑碰撞退化后的有效跳跃距离。

shift 表是一个以 BBB-gram 为键的映射，键空间大小为 ∣Σ∣B|\Sigma|^B∣Σ∣B。每条模式串 pip_ipi​ 向 shift 表贡献至多 ∣pi∣|p_i|∣pi​∣ 个 BBB-gram 键。

设 WM 侧模式串总字符数为：

SWM=∑pi∈PWM∣pi∣S_{\text{WM}} = \sum_{p_i \in \mathcal{P}_{\text{WM}}} |p_i| SWM​=pi​∈PWM​∑​∣pi​∣

shift 表中被占用的键总数上界为 K≤SWMK \leq S_{\text{WM}}K≤SWM​。当 KKK 接近或超过 ∣Σ∣B|\Sigma|^B∣Σ∣B 时，大量文本位置的 BBB-gram 都能命中 shift 表中 shift=0 的项，跳跃能力丧失，因此引入碰撞退化因子 ρ\rhoρ 来建模 shift 表饱和程度对跳跃距离的影响：

ρ(k)=min⁡ ⁣(1,  ∣Σ∣BSWM(k))\rho(k) = \min\!\left(1,\; \frac{|\Sigma|^B}{S_{\text{WM}}(k)}\right) ρ(k)=min(1,SWM​(k)∣Σ∣B​)

有效跳跃距离为：

deff(k)=max⁡ ⁣(1,  (mmin⁡(k)−B+1)⋅ρ(k))d_{\text{eff}}(k) = \max\!\left(1,\; (m_{\min}(k) - B + 1) \cdot \rho(k)\right) deff​(k)=max(1,(mmin​(k)−B+1)⋅ρ(k))

当 SWM≤∣Σ∣BS_{\text{WM}} \leq |\Sigma|^BSWM​≤∣Σ∣B 时 ρ=1\rho = 1ρ=1，shift 表无退化；当 SWM>∣Σ∣BS_{\text{WM}} > |\Sigma|^BSWM​>∣Σ∣B 时 ρ<1\rho < 1ρ<1，有效跳跃距离按比例缩短，WM 代价相应增大。

设分割点为 kkk，前 kkk 个模式串给 AC，后 N−kN-kN−k 个给 WM。定义总代价函数：

C(k)=max⁡ ⁣(TACprep(k),  TWMprep(k))⏟Tprep+max⁡ ⁣(TACmatch,  TWMmatch(k))⏟TmatchC(k) = \underbrace{\max\!\left(T_{\text{AC}}^{\text{prep}}(k),\; T_{\text{WM}}^{\text{prep}}(k)\right)}_{T_{\text{prep}}} + \underbrace{\max\!\left(T_{\text{AC}}^{\text{match}},\; T_{\text{WM}}^{\text{match}}(k)\right)}_{T_{\text{match}}} C(k)=Tprep​max(TACprep​(k),TWMprep​(k))​​+Tmatch​max(TACmatch​,TWMmatch​(k))​​

代入各项：

C(k)=max⁡ ⁣(∣Σ∣⋅SAC(k),    Ch⋅nWM⋅mmin⁡(k))+max⁡ ⁣(L,    Ch⋅Ldeff(k))C(k) = \max\!\left(|\Sigma| \cdot S_{\text{AC}}(k),\;\; C_h \cdot n_{\text{WM}} \cdot m_{\min}(k)\right) + \max\!\left(L,\;\; \frac{C_h \cdot L}{d_{\text{eff}}(k)}\right) C(k)=max(∣Σ∣⋅SAC​(k),Ch​⋅nWM​⋅mmin​(k))+max(L,deff​(k)Ch​⋅L​)

其中 SAC(k)=∑i=1k∣pi∣S_{\text{AC}}(k) = \sum_{i=1}^{k} |p_i|SAC​(k)=∑i=1k​∣pi​∣，nWM=N−kn_{\text{WM}} = N - knWM​=N−k，mmin⁡(k)=∣pk+1∣m_{\min}(k) = |p_{k+1}|mmin​(k)=∣pk+1​∣。

优化目标：

k∗=arg⁡min⁡0≤k≤NC(k)k^* = \arg\min_{0 \leq k \leq N} C(k) k∗=arg0≤k≤Nmin​C(k)

C(k)C(k)C(k) 的两个 max⁡\maxmax 项中涉及的各分量关于 kkk 的单调性不同（SACS_{\text{AC}}SAC​ 递增而 deffd_{\text{eff}}deff​ 也递增），C(k)C(k)C(k) 不具备全局单调性或凸性，因此无法直接二分搜索；因此预先构建后缀和 suffix(k)\text{suffix}(k)suffix(k)（O(N)O(N)O(N)），然后线性扫描所有 k∈[0,N]k \in [0, N]k∈[0,N]，逐一评估 C(k)C(k)C(k) 并取最小值。总时间复杂度 O(N)O(N)O(N)。

int main(int argc, char* argv[]) {
    cin.tie(0)->sync_with_stdio(0);

    cerr << "初始内存使用情况: " << get_memory_kb() << " KB\n";

    // 读入所有模式串
    int fd = open(argv[1], O_RDONLY);
    dup2(fd, STDIN_FILENO);
    close(fd);

    vector<string> patterns;
    string s;
    while (cin >> s) patterns.push_back(s);
    cerr << "模式串总数: " << patterns.size() << "\n";

    // 按长度升序排列（命题 2：WM 取最长模式串可最大化 m_min）
    sort(patterns.begin(), patterns.end(), [](const string& a, const string& b) {
        return a.size() < b.size();
    });

    // 提前读入文本以获取总文本长度 L（代价模型需要）
    fd = open(argv[2], O_RDONLY);
    dup2(fd, STDIN_FILENO);
    close(fd);
    cin.clear();

    vector<string> texts;
    string text;
    while (cin >> text) texts.push_back(text);

    long long L = 0;
    for (const string& t : texts) L += t.size();

    // === 代价模型：最小化 T_prep + T_match ===
    //
    // 分割点 k：前 k 个模式串给 AC，后 n-k 个给 WM。
    //
    // 预处理（并行化）：
    //   T_AC_prep ∝ |Σ| × S_AC(k)         AC build 对每个 trie 节点遍历 256 个转移
    //   T_WM_prep ∝ C_hash × n_wm × m_min  WM build 对每个模式串做 m_min 次 hash 插入
    //   T_prep = max(T_AC_prep, T_WM_prep)
    //
    // 匹配（并行化）：
    //   T_AC_match ∝ L                      AC 每字符 1 次数组访问
    //   T_WM_match ∝ C_hash × L / d_eff     WM 每步 1 次 hash 查找，步长 d_eff
    //   T_match = max(T_AC_match, T_WM_match)
    //
    // 其中 d_eff = (m_min - B + 1) × min(1, Σ^B / S_WM) 包含碰撞退化因子
    //
    // 目标：min_k { T_prep(k) + T_match(k) }

    long long n = patterns.size();

    long long SIGMA_B = 1;
    for (long long i = 0; i < WM::B; i++) SIGMA_B *= 256; // 256^B

    // 后缀长度和: suffix_len[i] = Σ|p_j| for j in [i, n)
    vector<long long> suffix_len(n + 1, 0);
    for (long long i = n - 1; i >= 0; i--)
        suffix_len[i] = suffix_len[i + 1] + (long long)patterns[i].size();

    long long total_len = suffix_len[0];

    long long n_ac;

    if (argc >= 4) {
        // 用户指定长度阈值：长度 < threshold 的给 AC，>= threshold 的给 WM
        long long threshold = atoll(argv[3]);
        // patterns 已按长度升序排列，找第一个 >= threshold 的位置
        n_ac = (long long)(lower_bound(patterns.begin(), patterns.end(), threshold,
            [](const string& p, long long thr) { return (long long)p.size() < thr; }
        ) - patterns.begin());
        cerr << "使用用户指定的长度阈值: " << threshold
             << " (长度<" << threshold << " 给AC, >=" << threshold << " 给WM)\n";
    } else {
        // 代价模型权重（相对值，以单次数组访问为 1 单位）
        const double SIGMA = 256.0;       // AC build: 每节点 256 次数组操作
        const double C_HASH = 40.0;       // hash 操作相对数组访问的代价倍数

        long long best_k = n; // 默认全给 AC
        double best_cost = 1e18;

        for (long long k = 0; k <= n; k++) {
            double s_ac = (double)(total_len - suffix_len[k]);
            double s_wm = (double)suffix_len[k];
            long long n_wm_k = n - k;

            // 预处理代价
            double ac_prep = SIGMA * s_ac;
            double wm_prep = 0;

            // 匹配代价
            double ac_match = (k > 0) ? (double)L : 0;
            double wm_match = 0;

            if (n_wm_k > 0) {
                long long m_min_k = (long long)patterns[k].size();
                wm_prep = C_HASH * (double)n_wm_k * (double)m_min_k;

                double d_raw = max(1.0, (double)(m_min_k - WM::B + 1));
                double sat = (s_wm > 0) ? min(1.0, (double)SIGMA_B / s_wm) : 1.0;
                double d_eff = max(1.0, d_raw * sat);
                wm_match = C_HASH * (double)L / d_eff;
            }

            double cost;
            if (k == 0) {
                cost = wm_prep + wm_match;
            } else if (n_wm_k == 0) {
                cost = ac_prep + ac_match;
            } else {
                cost = max(ac_prep, wm_prep) + max(ac_match, wm_match);
            }

            if (cost < best_cost) {
                best_cost = cost;
                best_k = k;
            }
        }

        n_ac = best_k;
    }
    long long n_wm = n - n_ac;

    cerr << "Σ^B=" << SIGMA_B
         << ", WM总长=" << suffix_len[n_ac]
         << ", AC模式串数=" << n_ac << ", WM模式串数=" << n_wm << "\n";

    // 按下标分割构建
    AC::reset();
    WM::reset();
    for (long long i = 0; i < n_ac; i++) AC::insert(patterns[i]);
    for (long long i = n_ac; i < n; i++) WM::pat.push_back({patterns[i], 0});
    WM::n = WM::pat.size();
    cerr << "AC 模式串数: " << n_ac << ", WM 模式串数: " << n_wm << "\n";

    // 并行预处理
    auto t0 = chrono::high_resolution_clock::now();
    if (n_ac > 0 && n_wm > 0) {
        thread ac_build_thread([&]() { AC::build(); });
        thread wm_build_thread([&]() { WM::build(); });
        ac_build_thread.join();
        wm_build_thread.join();
    } else if (n_ac > 0) {
        AC::build();
    } else {
        WM::build();
    }
    auto t1 = chrono::high_resolution_clock::now();

    cerr << "预处理时间花费: "
         << chrono::duration<double, milli>(t1 - t0).count() << " ms\n";
    cerr << "预处理后内存使用情况: " << get_memory_kb() << " KB\n";

    long long wm_m = WM::m;

    // 并行匹配
    auto t2 = chrono::high_resolution_clock::now();

    long long ac_res = 0, wm_res = 0;

    thread ac_thread([&]() {
        if (n_ac > 0)
            for (const string &t : texts)
                ac_res += AC::query(t);
    });

    thread wm_thread([&]() {
        if (n_wm > 0)
            for (const string &t : texts)
                if ((long long)t.size() >= wm_m)
                    wm_res += WM::query(t);
    });

    ac_thread.join();
    wm_thread.join();

    auto t3 = chrono::high_resolution_clock::now();

    cerr << "匹配时间花费: "
         << chrono::duration<double, milli>(t3 - t2).count() << " ms\n";
    cerr << "匹配后内存使用情况: " << get_memory_kb() << " KB\n";
    cerr << "AC 匹配结果: " << ac_res << ", WM 匹配结果: " << wm_res << "\n";
    cerr << "总匹配数: " << ac_res + wm_res << "\n";

    return 0;
}
