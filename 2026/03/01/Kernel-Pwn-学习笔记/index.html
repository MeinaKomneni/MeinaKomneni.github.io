调试

boot.sh 添加调试 (-gdb tcp::1234)：

#!/bin/bash
qemu-system-x86_64 -initrd rootfs.cpio -kernel bzImage -append 'console=ttyS0 root=/dev/ram oops=panic panic=1' -enable-kvm -monitor /dev/null -m 128M --nographic  -smp cores=1,threads=1 -cpu kvm64,+smep -gdb tcp::1234

如果是平时学习，可以自己编译出带符号的内核、自己写有洞的模块并测试，调试起来是非常舒服的；

而对于做题来说，则需要用 extract-vmlinux 提取 vmlinux，用 vmlinux_to_elf 恢复函数名，才能带函数名进行调试。

笔者更喜欢写两个文件： exp.c 和 exp.py，exp.c 是名副其实的exp，而 exp.py 用来打包、启动并调试；启动后Ctrl+c停在native_safe_halt处，接着在sys_execve处下断点，最后运行exp。

这里附上本人学习用的模块源码和脚本。

my_module.c

#include <linux/module.h>
#include <linux/kernel.h>
#include <linux/init.h>
#include <linux/fs.h>
#include <linux/cdev.h>
#include <linux/uaccess.h>
#include <linux/miscdevice.h>
#include <linux/cred.h>
#include <linux/sched.h>
#include <linux/mm.h>

#define DEVICE_NAME "debug_module"
#define IOCTL_GET_ROOT 0xdeadbeef
#define IOCTL_GET_PHYS 0xcafebabe
#define IOCTL_HEAP_OP 0xfeedface

struct phys_query {
    unsigned long vaddr;
    unsigned long long paddr;
};

struct heap_op {
    size_t idx;
    size_t op; // 0: alloc, 1: free
    size_t size; // for alloc
};
void* heap_list[16] = {0};

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Test Author");
MODULE_DESCRIPTION("A kernel module for debug, DONT USE IN PRODUCTION");

static long do_get_root(void)
{
    struct cred *new_creds;

    new_creds = prepare_creds();

    if (!new_creds) {
        printk(KERN_ERR "[debug] Failed to prepare new credentials\n");
        return -ENOMEM;
    }
    
    new_creds->uid.val = new_creds->gid.val = 0;
    new_creds->euid.val = new_creds->egid.val = 0;
    new_creds->suid.val = new_creds->sgid.val = 0;
    new_creds->fsuid.val = new_creds->fsgid.val = 0;

    commit_creds(new_creds);
    printk(KERN_INFO "[debug] root! granted to PID %d\n", current->pid);
    return 0;
}

static long do_get_phys(unsigned long arg)
{
    struct phys_query req;
    struct page *page = NULL;
    int res;

    if (copy_from_user(&req, (void __user *)arg, sizeof(req))){
        printk(KERN_ERR "[debug] Failed to copy data from user\n");
        return -EFAULT;
    }

    if (res = get_user_pages_fast(req.vaddr, 1, 0, &page)) {
        printk(KERN_ERR "[debug] get_user_pages_fast failed for 0x%lx: %d\n", req.vaddr, res);
        return -EFAULT;
    }

    req.paddr = page_to_phys(page) + (req.vaddr & ~PAGE_MASK);
    put_page(page); 

    if (copy_to_user((void __user *)arg, &req, sizeof(req))){
        printk(KERN_ERR "[debug] Failed to copy data to user\n");
        return -EFAULT;
    }

    printk(KERN_INFO "[debug] vaddr: 0x%lx -> paddr: 0x%llx\n", req.vaddr, req.paddr);
    return 0;
}

static long do_heap_op(unsigned long arg)
{
    struct heap_op req;
    if (copy_from_user(&req, (void __user *)arg, sizeof(req))){
        printk(KERN_ERR "[debug] Failed to copy data from user\n");
        return -EFAULT;
    }
    switch (req.op){
        case 1: // alloc
            if (heap_list[req.idx]) {
                printk(KERN_ERR "[debug] Heap slot %zu already allocated\n", req.idx);
                return -EINVAL;
            }
            heap_list[req.idx] = kmalloc(req.size, GFP_KERNEL);
            if (!heap_list[req.idx]) {
                printk(KERN_ERR "[debug] Failed to allocate heap slot %zu\n", req.idx);
                return -ENOMEM;
            }
            printk(KERN_INFO "[debug] Allocated %zu bytes at heap slot %zu\n", req.size, req.idx);
            break;
        case 2: // free
            if (!heap_list[req.idx]) {
                printk(KERN_ERR "[debug] Heap slot %zu is not allocated\n", req.idx);
                return -EINVAL;
            }
            kfree(heap_list[req.idx]);
            printk(KERN_INFO "[debug] Freed heap slot %zu\n", req.idx);
            break;
        case 3: //edit
            if (!heap_list[req.idx]) {
                printk(KERN_ERR "[debug] Heap slot %zu is not allocated\n", req.idx);
                return -EINVAL;
            }
            if (copy_from_user(heap_list[req.idx], (void __user *)arg + sizeof(req), req.size)){
                printk(KERN_ERR "[debug] Failed to copy data from user to heap slot %zu\n", req.idx);
                return -EFAULT;
            }
            printk(KERN_INFO "[debug] Edited heap slot %zu with %zu bytes\n", req.idx, req.size);
            break;
        case 4: // show
            if (!heap_list[req.idx]) {
                printk(KERN_ERR "[debug] Heap slot %zu is not allocated\n", req.idx);
                return -EINVAL;
            }
            if (copy_to_user((void __user *)arg + sizeof(req), heap_list[req.idx], req.size)){
                printk(KERN_ERR "[debug] Failed to copy data from heap slot %zu to user\n", req.idx);
                return -EFAULT;
            }
            printk(KERN_INFO "[debug] Showed heap slot %zu with %zu bytes\n", req.idx, req.size);
            break;
        default:
            printk(KERN_ERR "[debug] Invalid heap operation: %zu\n", req.op);
            return -EINVAL;
    } 
    return 0;
}

static long my_ioctl(struct file *file, unsigned int cmd, unsigned long arg)
{
    switch (cmd) {
        case IOCTL_GET_ROOT:
            return do_get_root();
        case IOCTL_GET_PHYS:
            return do_get_phys(arg);
        case IOCTL_HEAP_OP:
            return do_heap_op(arg);
        default:
            break;
    }
    return -EINVAL;
}

static const struct file_operations debug_fops = {
    .owner = THIS_MODULE,
    .unlocked_ioctl = my_ioctl,
};

static struct miscdevice debug_misc = {
    .minor = MISC_DYNAMIC_MINOR,
    .name = DEVICE_NAME,
    .fops = &debug_fops,
};

static int __init debug_init(void)
{
    int ret;

    ret = misc_register(&debug_misc);
    if (ret) {
        printk(KERN_ERR "[debug] Failed to register misc device\n");
        return ret;
    }

    printk(KERN_INFO "[debug] Module loaded. Device: /dev/%s\n", DEVICE_NAME);
    return 0;
}

static void __exit debug_exit(void)
{
    misc_deregister(&debug_misc);
    printk(KERN_INFO "[debug] Module unloaded\n");
}

module_init(debug_init);
module_exit(debug_exit);

void dump(unsigned char* buf, int len) {

exp.py

#!/usr/bin/env python3
import os
import sys
import shutil
from pwn import *

# making .bak

bzImage = "./bzImage"
fs = "./rootfs.cpio"
vmlinux = "~/pwn/kernel/linux-6.18.5/vmlinux"

context.log_level = 'debug'
context.terminal = ['tmux', 'splitw', '-h']
context.arch = 'amd64'

if not os.path.exists("./bzImage.bak"):
    shutil.copy(bzImage, "./bzImage.bak")

if not os.path.exists("./rootfs.cpio.bak"):
    shutil.copy(fs, "./rootfs.cpio.bak")

if args.GDB:
    # compile exp.c
    os.system("gcc -o exp exp.c -g --static && chmod +777 exp")
    # cp
    if not os.path.exists("./fs"):
        raise Exception("fs directory not found")
    os.makedirs("./fs/tmp", exist_ok=True)
    shutil.copy("./exp", "./fs/tmp/exp")
    # make cpio
    os.system("cd fs && find . | cpio -o -H newc > ../rootfs.cpio")
    # run qemu
    io = process("./boot.sh")
    
    # start gdb
    gdbscript = f'''
    target remote :1234
    '''
    
    # Write gdbscript to a temporary file
    with open("/tmp/gdbscript", "w") as f:
        f.write(gdbscript)
        
    # Launch gdb in a new terminal using tmux
    os.system(f"tmux splitw -h 'gdb -x /tmp/gdbscript {vmlinux}'")
    
    io.interactive()

if args.REMOTE:
    # compile exp.c
    context.log_level = 'info'
    os.system("musl-gcc -o exp exp.c -static -Os -s -fno-ident -fno-asynchronous-unwind-tables -Wl,--build-id=none && chmod +777 exp")

    with open("exp", "rb") as f:
        con = f.read()
    
    pay = base64.b64encode(con).decode()
    io = conn = remote('101.42.41.80', 32864)
    io.sendline()
    io.recvuntil(b'$')
    io.sendline(b'cd /tmp')
    chunk_size = 800
    total = len(pay)

    for i in range(0, total, chunk_size):
        chunk = pay[i:i+chunk_size]
        io.sendline(f'echo "{chunk}" >> exp.b64'.encode())
        time.sleep(0.05)
        if i % (chunk_size * 5) == 0:
            log.info(f"Progress: {i}/{total} bytes")

    io.sendline(b'base64 -d exp.b64 > exp')
    io.sendline(b'chmod +x exp')
    io.sendline(b'rm exp.b64')
    log.success("File uploaded successfully!")
    io.sendline(b'./exp')

    io.interactive()
